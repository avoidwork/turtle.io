src/hash.js

Method hash

Creates a hash of arg

Parameters:

  • arg can be of any type.
    (String or Buffer)

  • encrypt must be a String.
    ([Optional] Type of encryption)

  • digest must be a String.
    ([Optional] Type of digest)

Returns a String
(Hash of arg)

TurtleIO.prototype.hash = function ( arg, encrypt, digest ) { encrypt = encrypt || "md5"; digest = digest || "hex"; if ( typeof arg !== "string" && !arg instanceof Buffer ) { arg = ""; } return crypto.createHash( encrypt ).update( arg ).digest( digest ); };
